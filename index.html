export function PulseOrb({ reps, active, onTap }) {
  const state = useRef({ pulse: 0, repRate: 0 });

  useEffect(() => {
    // visual burst on every rep
    s.pulse = 1.0;
    s.rotationSpeed = 0.003 + s.repRate * 0.003;
  }, [reps]);

  function draw() {
    const hue = 150 + s.repRate * 40;
    for (let layer = 4; layer >= 0; layer--) {
      const grad = ctx.createRadialGradient(0,0,0,0,0,R);
      grad.addColorStop(0, `hsla(${hue},90%,70%,.85)`);
      ctx.fill();
    }
    requestAnimationFrame(draw);
  }
}
